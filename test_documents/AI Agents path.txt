def make_page_template(canvas, doc):
    # White background
    canvas.saveState()
    canvas.setFillColor(WHITE)
    canvas.rect(0, 0, W, H, fill=1, stroke=0)
    # Bottom footer line
    canvas.setStrokeColor(DIVIDER)
    canvas.setLineWidth(0.5)
    canvas.line(20*mm, 14*mm, W - 20*mm, 14*mm)
    # Footer text
    canvas.setFont('Helvetica', 8)
    canvas.setFillColor(MID_GRAY)
    canvas.drawString(20*mm, 10*mm, "50 Days to AI Engineer — Production-Grade Project Roadmap")
    canvas.drawRightString(W - 20*mm, 10*mm, f"Page {doc.page}")
    canvas.restoreState()

def phase_banner(text, phase_num):
    col = PHASE_COLORS[phase_num]
    data = [[Paragraph(text, section_header)]]
    t = Table(data, colWidths=[W - 40*mm])
    t.setStyle(TableStyle([
        ('BACKGROUND', (0,0), (-1,-1), col),
        ('TOPPADDING', (0,0), (-1,-1), 10),
        ('BOTTOMPADDING', (0,0), (-1,-1), 10),
        ('LEFTPADDING', (0,0), (-1,-1), 14),
        ('RIGHTPADDING', (0,0), (-1,-1), 14),
        ('ROWBACKGROUNDS', (0,0), (-1,-1), [col]),
    ]))
    return t

        ('RIGHTPADDING', (0,0), (-1,-1), 8),
        ('ROWBACKGROUNDS', (0,0), (-1,-1), [bg_color]),

    return t
